<script>

  // Password access logic

  const correctKey = "77185";

  const preAccess = document.getElementById('preAccess');

  const keyInput = document.getElementById('keyInput');

  const content = document.getElementById('content');

  keyInput.addEventListener('keydown', function(e) {

    if (e.key === 'Enter') {

      if (keyInput.value === correctKey) {

        // Unlock the site

        preAccess.classList.add('hidden');

        setTimeout(() => {

          preAccess.style.display = 'none';

        }, 500);

        // Remove blur

        document.getElementById('content').classList.remove('blurred');

      } else {

        alert('Incorrect key. Try again.');

        keyInput.value = '';

      }

    }

  });

  // Bubbles animation

  const bubblesContainer = document.getElementById('bubbles');

  function createBubble() {

    const bubble = document.createElement('div');

    const size = Math.random() * 10 + 10; // size between 10-20px

    bubble.style.width = size + 'px';

    bubble.style.height = size + 'px';

    bubble.style.background = 'rgba(0, 255, 255, 0.5)';

    bubble.style.borderRadius = '50%';

    bubble.style.position = 'absolute';

    bubble.style.bottom = '0px';

    bubble.style.left = Math.random() * 100 + '%';

    bubble.style.opacity = Math.random();

    bubble.style.transform = 'translateY(0)';

    bubble.style.zIndex = 0;

    bubblesContainer.appendChild(bubble);

    // Animate bubble

    const duration = Math.random() * 5 + 5; // seconds

    const translateY = - (Math.random() * 300 + 100); // pixels

    bubble.animate([

      { transform: 'translateY(0)', opacity: bubble.style.opacity },

      { transform: `translateY(${translateY}px)`, opacity: 0 }

    ], {

      duration: duration * 1000,

      easing: 'linear',

      fill: 'forwards'

    }).onfinish = () => {

      bubble.remove();

    };

  }

  // Generate bubbles continuously

  setInterval(createBubble, 300);

  // Optional: unlock the site visually

  // Initially, content is not blurred; blur when preAccess appears

  // Since you requested "around untouchable and blurred," we can add blur

  document.getElementById('content').classList.add('blurred');

</script>
